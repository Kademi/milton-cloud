20  package io.milton.resource;
21  
22  import io.milton.http.LockInfo;
23  import io.milton.http.LockResult;
24  import io.milton.http.LockTimeout;
25  import io.milton.http.LockToken;
26  import io.milton.resource.Resource;
27  import io.milton.http.exceptions.LockedException;
28  import io.milton.http.exceptions.NotAuthorizedException;
29  import io.milton.http.exceptions.PreConditionFailedException;
30  
31  /**
32   * webDAV LOCK
33   *
34   * You should also implement LockingCollectionResource on your collections for full
35   * locking support
36   * 
37   * @author brad
38   */
39  public interface LockableResource extends Resource {
40      /**
41       * Lock this resource and return a token
42       * 
43       * @param timeout - in seconds, or null
44       * @param lockInfo
45       * @return - a result containing the token representing the lock if succesful,
46       * otherwise a failure reason code
47       */
48      public LockResult lock(LockTimeout timeout, LockInfo lockInfo) throws NotAuthorizedException, PreConditionFailedException, LockedException;
49      
50      /**
51       * Renew the lock and return new lock info
52       * 
53       * @param token
54       * @return
55       */
56      public LockResult refreshLock(String token) throws NotAuthorizedException, PreConditionFailedException;
57  
58      /**
59       * If the resource is currently locked, and the tokenId  matches the current
60       * one, unlock the resource
61       *
62       * @param tokenId
63       */
64      public void unlock(String tokenId) throws NotAuthorizedException, PreConditionFailedException;
65  
66      /**
67       *
68       * @return - the current lock, if the resource is locked, or null
69       */
70      public LockToken getCurrentLock();
71  }
