19  package io.milton.common;
20  
21  import java.io.File;
22  import java.util.ArrayList;
23  import java.util.Arrays;
24  import java.util.List;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  /**
29   *
30   * @author brad
31   */
32  public class ContentTypeUtils {
33  
34      private static Logger log = LoggerFactory.getLogger(ContentTypeUtils.class);
35      private static final ContentTypeService contentTypeService = new DefaultContentTypeService(); // will load props file ;
36  
37  
38      public static String findContentTypes(String name) {
39          List<String> list = contentTypeService.findContentTypes(name);
40          return buildContentTypeText(list);
41      }
42  
43      public static String findContentTypes(File file) {
44          return buildContentTypeText(contentTypeService.findContentTypes(file.getName()));
45      }
46  
47      public static String findAcceptableContentType(String canProvide, String accepts) {
48          return contentTypeService.getPreferedMimeType(accepts, toList(canProvide));
49      }
50  
51      public static String findAcceptableContentTypeForName(String name, String accepts) {
52          String canProvide = findContentTypes(name);
53          List<String> canProvideList = toList(canProvide);
54          String s = contentTypeService.getPreferedMimeType(accepts, canProvideList);
55          return s;
56      }
57      
58      
59      private static String buildContentTypeText(List<String> mimeTypes) {
60          return Utils.toCsv(mimeTypes);
61      }
62  
63      public static List<String> toList(String s) {
64          List<String> list = new ArrayList<String>();
65          if( s != null ) {
66              for( String x : s.split(",")) {
67                  x = x.trim();
68                  list.add(x);
69              }
70          }
71          return list;
72      }
73  }
