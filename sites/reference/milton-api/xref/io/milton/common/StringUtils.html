30  public abstract class StringUtils {
31  
32      public static String[] delimitedListToStringArray( String str, String delimiter ) {
33          if( str == null ) {
34              return new String[0];
35          }
36          if( delimiter == null ) {
37              return new String[]{str};
38          }
39          List result = new ArrayList();
40          if( "".equals( delimiter ) ) {
41              for( int i = 0; i < str.length(); i++ ) {
42                  result.add( str.substring( i, i + 1 ) );
43              }
44          } else {
45              int pos = 0;
46              int delPos = 0;
47              while( ( delPos = str.indexOf( delimiter, pos ) ) != -1 ) {
48                  result.add( str.substring( pos, delPos ) );
49                  pos = delPos + delimiter.length();
50              }
51              if( str.length() > 0 && pos <= str.length() ) {
52                  // Add rest of String, but not in case of empty input.
53                  result.add( str.substring( pos ) );
54              }
55          }
56          return toStringArray( result );
57      }
58  
59      public static String[] toStringArray( Collection collection ) {
60          if( collection == null ) {
61              return null;
62          }
63          return (String[]) collection.toArray( new String[collection.size()] );
64      }

78  			}
