20  package io.milton.http;
21  
22  /**
23   *
24   * @author brad
25   */
26  public interface Cookie {
27  
28  
29     /**
30      * This returns the version for this cookie. The version is
31      * not optional and so will always return the version this
32      * cookie uses. If no version number is specified this will
33      * return a version of 1, to comply with RFC 2109.
34      *
35      * @return the version value from this cookie instance
36      */
37     int getVersion();
38  
39     /**
40      * This enables the version of the <code>Cookie</code> to be
41      * set. By default the version of the <code>Cookie</code> is
42      * set to 1. It is not advisable to set the version higher
43      * than 1, unless it is known that the client will accept it.
44      * <p>
45      * Some old browsers can only handle cookie version 0. This
46      * can be used to comply with the original Netscape cookie
47      * specification. Version 1 complies with RFC 2109.
48      *
49      * @param version this is the version number for the cookie
50      */
51     void setVersion(int version);
52  
53     /**
54      * This returns the name for this cookie. The name and value
55      * attributes of a cookie define what the <code>Cookie</code>
56      * is for, these values will always be present. These are
57      * mandatory for both the Cookie and Set-Cookie headers.
58      * <p>
59      * Because the cookie may be stored by name, the cookie name
60      * cannot be modified after the creation of the cookie object.
61      *
62      * @return the name from this cookie instance object
63      */
64     String getName();
65  
66     /**
67      * This returns the value for this cookie. The name and value
68      * attributes of a cookie define what the <code>Cookie</code>
69      * is for, these values will always be present. These are
70      * mandatory for both the Cookie and Set-Cookie headers.
71      *
72      * @return the value from this cookie instance object
73      */
74     String getValue();
75  
76     /**
77      * This enables the value of the cookie to be changed. This
78      * can be set to any value the server wishes to send. Cookie
79      * values can contain space characters as they are transmitted
80      * in quotes. For example a value of <code>some value</code>
81      * is perfectly legal. However for maximum compatibility
82      * across the different plaforms such as PHP, JavaScript and
83      * others, quotations should be avoided. If quotations are
84      * required they must be added to the string. For example a
85      * quoted value could be created as <code>"some value"</code>.
86      *
87      * @param value this is the new value of this cookie object
88      */
89     void setValue(String value);
90  
91     /**
92      * This determines whether the cookie is secure. The cookie
93      * is secure if it has the "secure" token set, as defined
94      * by RFC 2109. If this token is set then the cookie is only
95      * sent over secure channels such as SSL and TLS and ensures
96      * that a third party cannot intercept and spoof the cookie.
97      *
98      * @return this returns true if the "secure" token is set
99      */
100    boolean getSecure();
101 
102    /**
103     * This is used to determine if the client browser should send
104     * this cookie over a secure protocol. If this is true then
105     * the client browser should only send the cookie over secure
106     * channels such as SSL and TLS. This ensures that the value
107     * of the cookie cannot be intercepted by a third party.
108     *
109     * @param secure if true then the cookie should be protected
110     */
111    void setSecure(boolean secure);
112 
113    /**
114     * This returns the number of seconds a cookie lives for. This
115     * determines how long the cookie will live on the client side.
116     * If the expiry is less than zero the cookie lifetime is the
117     * duration of the client browser session, if it is zero then
118     * the cookie will be deleted from the client browser.
119     *
120     * @return returns the duration in seconds the cookie lives
121     */
122    int getExpiry();
123 
124    /**
125     * This allows a lifetime to be specified for the cookie. This
126     * will make use of the "max-age" token specified by RFC 2109
127     * the specifies the number of seconds a browser should keep
128     * a cookie for. This is useful if the cookie is to be kept
129     * beyond the lifetime of the client session. If the valie of
130     * this is zero then this will remove the client cookie, if
131     * it is less than zero then the "max-age" field is ignored.
132     *
133     * @param expiry the duration in seconds the cookie lives
134     */
135    void setExpiry(int expiry);
136 
137    /**
138     * This returns the path for this cookie. The path is in both
139     * the Cookie and Set-Cookie headers and so may return null
140     * if there is no domain value. If the <code>toString</code>
141     * or <code>toClientString</code> is invoked the path will
142     * not be present if the path attribute is null.
143     *
144     * @return this returns the path value from this cookie
145     */
146    String getPath();
147 
148    /**
149     * This is used to set the cookie path for this cookie. This
150     * is set so that the cookie can specify the directories that
151     * the cookie is sent with. For example if the path attribute
152     * is set to <code>/pub/bin</code>, then requests for the
153     * resource <code>http://hostname:port/pub/bin/README</code>
154     * will be issued with this cookie. The cookie is issued for
155     * all resources in the path and all subdirectories.
156     *
157     * @param path this is the path value for this cookie object
158     */
159    void setPath(String path);
160 
161    /**
162     * This returns the domain for this cookie. The domain is in
163     * both the Cookie and Set-Cookie headers and so may return
164     * null if there is no domain value. If either the
165     * <code>toString</code> or <code>toClientString</code> is
166     * invoked the domain will not be present if this is null.
167     *
168     * @return this returns the domain value from this cookie
169     */
170    String getDomain();
171 
172    /**
173     * This enables the domain for this <code>Cookie</code> to be
174     * set. The form of the domain is specified by RFC 2109. The
175     * value can begin with a dot, like <code>.host.com</code>.
176     * This means that the cookie is visible within a specific
177     * DNS zone like <code>www.host.com</code>. By default this
178     * value is null which means it is sent back to its origin.
179     *
180     * @param domain this is the domain value for this cookie
181     */
182    void setDomain(String domain);
183 
184 }
