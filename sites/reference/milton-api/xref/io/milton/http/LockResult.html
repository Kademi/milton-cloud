20  package io.milton.http;
21  
22  import io.milton.http.Response;
23  
24  /**
25   *
26   */
27  public class LockResult {
28  
29      public static LockResult failed( FailureReason failureReason) {
30          return new LockResult(failureReason, null);
31      }
32  
33      public static LockResult success(LockToken token) {
34          return new LockResult(null, token);
35      }
36  
37      public enum FailureReason {
38          ALREADY_LOCKED(Response.Status.SC_CONFLICT),
39          PRECONDITION_FAILED(Response.Status.SC_LOCKED);
40  
41          public Response.Status status;
42  
43          FailureReason(Response.Status status) {
44              this.status = status;
45          }
46      }
47  
48      final FailureReason failureReason;
49      final LockToken lockToken;
50  
51      public LockResult(FailureReason failureReason, LockToken lockToken) {
52          this.failureReason = failureReason;
53          this.lockToken = lockToken;
54      }
55  
56      public boolean isSuccessful() {
57          return failureReason == null;
58      }
59  
60      public FailureReason getFailureReason() {
61          return failureReason;
62      }
63  
64      public LockToken getLockToken() {
65          return lockToken;
66      }
67  
68      
69  }
