23  /**
24   *  Indicates that the current user is not able to perform the requested operation
25   *
26   * This should not normally be used. Instead, a resource should determine if
27   * a user can perform an operation in its authorised method
28   *
29   * However, this exception allows for cases where the authorised status can
30   * only be determined during processing
31   */
32  public class NotAuthorizedException extends MiltonException{
33      private static final long serialVersionUID = 1L;
34  
35      public NotAuthorizedException(Resource r) {
36          super(r);
37      }
