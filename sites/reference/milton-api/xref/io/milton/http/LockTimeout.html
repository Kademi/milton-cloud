37      private static Logger log = LoggerFactory.getLogger(LockTimeout.class);
38      private static final String INFINITE = "Infinite";
39      
40              
41      public static LockTimeout parseTimeout(Request request) {
42          String sTimeout = request.getTimeoutHeader();
43          log.debug("..requested timeout: " + sTimeout);
44          return parseTimeout(sTimeout);
45      }
46      
47      public static LockTimeout parseTimeout(String s) {      
48          if ( s==null ) return new LockTimeout((List<Long>)null);
49          s = s.trim();
50          if( s.length() == 0 ) return new LockTimeout((List<Long>)null);
51  
52          List<Long> list = new ArrayList<Long>();
53          for( String part : s.split(",")) {
54              part = part.trim();
55              if( part.equalsIgnoreCase(INFINITE)) {
56                  list.add(Long.MAX_VALUE);
57              } else {
58                  Long seconds = parseTimeoutPart(part);
59                  if(seconds != null ) {
60                      list.add(seconds);
61                  }
62              }
63          }
64          
65          LockTimeout timeout = new LockTimeout(list);
66          return timeout;
67      }
68  
69      static String trim(String s) {
70          if( s == null ) return "";
71          return s.trim();
72      }
73      
74      static boolean isPresent(String s) {
75          return s != null && s.length()>0;
76      }

78      private static Long parseTimeoutPart(String part) {
79          if( part == null || part.length() == 0 ) return null;
80          int pos = part.indexOf("-");
81          if( pos <= 0 ) {
82              return null;
83          }
84          String s = part.substring(pos+1, part.length());
85          long l = 0;
86          try {
87              l = Long.parseLong(s);
88              return l;
89          } catch (NumberFormatException numberFormatException) {
90              log.error("Number format exception parsing timeout: " + s);
91              return null;
92          }        
93      }

95      final Long seconds;
96      final Long[] otherSeconds;
97  
98      public LockTimeout(Long timeout) {
99          this.seconds = timeout;
100         this.otherSeconds = null;
101     }
102     
103     private LockTimeout(List<Long> timeouts) {
104         if( timeouts == null || timeouts.isEmpty() ) {
105             this.seconds = null;
106             this.otherSeconds = null;
107         } else {
108             this.seconds = timeouts.get(0);
109             timeouts.remove(0);
110             otherSeconds = new Long[timeouts.size()];
111             timeouts.toArray(otherSeconds);
112         }
113     }

131     @Override

133         if( this.seconds == null ) {
134             return INFINITE;
135         } else if(this.seconds.equals( Long.MAX_VALUE )) {

145     /**
146      * Returns a current object which holds the expected end date/time, based
147      * on defaultSeconds and maxSeconds, as well as the actual seconds used
148      * in that calculation.
149      *
150      * This is handy for locking because we generally want to lock a resource
151      * until a specific date/time, but we also want to report back the timeout
152      * actually locked in terms of seconds
153      * 
154      * @return - the current time + getSeconds()
155      */
156     public DateAndSeconds getLockedUntil(Long defaultSeconds, Long maxSeconds) {
157         Long l = getSeconds();        
158         if( l == null ) {
159             if( defaultSeconds != null ) {
160                 return addSeconds(defaultSeconds);
161             } else if( maxSeconds != null ) {
162                 return addSeconds( maxSeconds);
163             } else {
164                 return addSeconds(60l); // default default
165             }
166         } else {
167             if( maxSeconds != null ) {
168                 if( getSeconds() > maxSeconds ) {
169                     return addSeconds(maxSeconds);
170                 } else {
171                     return addSeconds(l);
172                 }
173             } else {
174                 return addSeconds(l);
175             }
176         }
177     }
178     
179     public static DateAndSeconds addSeconds( Long l) {
180         Calendar cal = Calendar.getInstance();
181         cal.setTimeInMillis(System.currentTimeMillis());
182         int secs = (int)l.longValue();
183         cal.add(Calendar.SECOND, secs);
184         DateAndSeconds das = new DateAndSeconds();
185         das.date = cal.getTime();        
186         das.seconds = l;
187         return das;
188     }
189         
190     public static class DateAndSeconds {
191         public Date date;
192         public Long seconds;
193     }
