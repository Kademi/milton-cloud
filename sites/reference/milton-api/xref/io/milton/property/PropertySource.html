20  package io.milton.property;
21  
22  import io.milton.resource.Resource;
23  import io.milton.http.Response;
24  import io.milton.http.Response.Status;
25  import io.milton.http.exceptions.BadRequestException;
26  import io.milton.http.exceptions.NotAuthorizedException;
27  import java.util.List;
28  import javax.xml.namespace.QName;
29  
30  /**
31   * Defines a source of properties. This is used by both propfind and proppatch
32   * 
33   *
34   * @author brad
35   */
36  public interface PropertySource {
37  
38      public enum PropertyAccessibility {
39  
40          UNKNOWN,
41          READ_ONLY,
42          WRITABLE
43      }
44  
45      public static class PropertyMetaData {
46  
47          private final PropertyAccessibility accessibility;
48          private final Class valueType;
49          public static final PropertyMetaData UNKNOWN = new PropertyMetaData( PropertyAccessibility.UNKNOWN, null );
50  
51          public PropertyMetaData( PropertyAccessibility accessibility, Class valueType ) {
52              this.accessibility = accessibility;
53              this.valueType = valueType;
54          }
55  
56          public PropertyAccessibility getAccessibility() {
57              return accessibility;
58          }
59  
60          public Class getValueType() {
61              return valueType;
62          }
63  
64          public boolean isUnknown() {
65              return accessibility.equals( PropertyAccessibility.UNKNOWN );
66          }
67  
68          public boolean isWritable() {
69              return accessibility.equals( PropertyAccessibility.WRITABLE );
70          }
71      }
72  
73  	/**
74  	 * Return the typed value of the given property. For example, if the property
75  	 * is a date/time value it should generally return a java.util.Date. Milton
76  	 * will take care of formatting types values into XML responses.
77  	 * 
78  	 * If the current user cannot read the requested property throw a NotAuthorizedException,
79  	 * this will set the appropriate status against this 
80  	 * 
81  	 * @param name
82  	 * @param r
83  	 * @return
84  	 * @throws NotAuthorizedException 
85  	 */
86      Object getProperty( QName name, Resource r ) throws NotAuthorizedException;
87  
88  	/**
89  	 * Update the given property to the value given. If the value is invalid
90  	 * throw a PropertySetException, and if the user is not permitted
91  	 * to update the property throw a NotAuthorizedException
92  	 * 
93  	 * @param name
94  	 * @param value
95  	 * @param r
96  	 * @throws com.bradmcevoy.property.PropertySource.PropertySetException
97  	 * @throws NotAuthorizedException 
98  	 */
99      void setProperty( QName name, Object value, Resource r ) throws PropertySetException, NotAuthorizedException;
100 
101     /**
102      * Check to see if the property is known, and if it is writable.
103      *
104      * The returned value also contains a class which is the most specific known
105      * class of the values which can be contained in this property. This class
106      * must be sufficient to locate a ValueWriter to parse the textual representation
107      * sent in PROPPATCH requests.
108      *
109      * @param name - the qualified name of the property
110      * @param r - the resource which might contain the property
111      * @return - never null, contains an enum value indicating if the property is known
112      * to this source, and if it is writable, and a class indicating the type of the property.
113      */
114     PropertyMetaData getPropertyMetaData( QName name, Resource r )throws NotAuthorizedException, BadRequestException;
115 
116     /**
117      * Remove the given property. There may be a semantic difference in some
118      * cases between setting a property to a null value vs removing the property.
119      * Generally this should completely the remove the property if possible.
120      *
121      * @param name
122      * @param r
123      */
124     void clearProperty( QName name, Resource r ) throws PropertySetException, NotAuthorizedException;
125 
126     /**
127      *
128      * @param r - the resource which may contain properties
129      * 
130      * @return - all properties known by this source on the given resource.
131      * This list should be exclusive. Ie only return properties not returned
132      * by any other source
133      */
134     List<QName> getAllPropertyNames( Resource r ) throws NotAuthorizedException, BadRequestException;
135 
136     /**
137      * Exception from setting a field
138      */
139     public class PropertySetException extends RuntimeException {
140 
141         private static final long serialVersionUID = 1L;
142         private Response.Status status;
143         private String notes;
144 
145         public PropertySetException( Status status, String notes ) {
146             this.status = status;
147             this.notes = notes;
148         }
149 
150         /**
151          * A response code indicating the problem. Eg 500 for a processing error,
152          * or conflict, forbidden, etc
153          *
154          * @return
155          */
156         public Status getStatus() {
157             return status;
158         }
159 
160         /**
161          * Should contain a human readable description of the problem
162          *
163          * @return
164          */
165         public String getErrorNotes() {
166             return notes;
167         }
168     }
169 }
