20  package io.milton.principal;
21  
22  import javax.xml.namespace.QName;
23  
24  /**
25   * Marker interface to identify those classes which can act as a Principal. This can be a user or a group
26   * 
27   * An absolute minimum requirement for a Principal is to be able to identify it, and
28   * this is represented by the sole method on this interface which is getIdentifier.
29   * 
30   * Note that some Principals are defined by the webdav protocol so must be subject
31   * to explicit processing requirements. For this reason the principle identifier
32   * is a structured type which can be associated with a namespace and well known
33   * names within that namespace.
34   * 
35   *
36   * There are 2 types of Principal
37   *  - Discrete: identifies a particular resource, which might be a user or group. Users are always discrete, groups may be
38   * physical resources in which case they are represented as discrete resources. But groups may also be dynamically evaluated..
39   *  - DAV: a dynamically evaluated group as defined by the ACL spec. Eg the Authenticated group matches any user
40   * with an authenticated request. A dynamically evaluated group defined by the protocol is not
41   * a physical resource do does not have a corresponding href
42   *
43   * @author brad
44   */
45  public interface Principal {
46  
47      /**
48       * A means to identify the principle to webdav clients
49       *
50       * @return
51       */
52      PrincipleId getIdenitifer();
53  
54      /**
55  	 * A principleId represents a unique identifier for a principle. It consists
56  	 * of 2 parts;
57  	 *  - type
58  	 *  - value (optional)
59  	 * 
60  	 * The type says this identifier identifies a physical resource or some construct
61  	 * defined by the ACL protocol. If the principal is a physical resource the
62  	 * type will be a Qualified Name with a namespace of D: (ie webdav) and a local
63  	 * name of href (which means the id is a href).
64  	 * 
65  	 * If the identifier identifies a protocol contruct the type will identify
66  	 * which construct (which must be known to the protocol and hence to milton), such
67  	 * as D:all, meaning all users
68  	 * 
69       * Eg
70       * <D:href>http://www.example.com/acl/groups/maintainers</D:href>
71       * For a specific user or group
72       *
73       * or
74       *
75       * <D:all/>  - for a group which represents all authenticated users, and is defined
76       * by the DAV standard
77       *
78       */
79      public static interface PrincipleId {
80          /**
81           * Eg D:href or D:all
82           *
83           * @return
84           */
85          QName getIdType();
86  
87          /**
88           * Eg null for an idType of "D:all", or http://blah.com/users/sam for "D:href"
89           *
90           * @return
91           */
92          String getValue();
93      }
94  }
